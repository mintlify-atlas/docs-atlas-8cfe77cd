## Overview

The multiplayer room system enables players to create and join real-time Moche game sessions with configurable difficulty, privacy settings, and automatic bot filling.

## Room Manager Architecture

The `RoomManager` class handles all room lifecycle operations (`moche_engine.py:4-242`):

```python
class RoomManager:
    def __init__(self):
        # dict mapping room_id to room data
        self.rooms = {}
    
    def create_room(self, host_id, host_name, is_private, bet_amount, total_slots, difficulty):
        room_id = str(uuid.uuid4())[:8] # Short UUID for easy sharing
        
        room = {
            "id": room_id,
            "host": host_id,
            "host_name": host_name,
            "is_private": is_private,
            "bet_amount": bet_amount,
            "total_slots": total_slots,
            "difficulty": difficulty,
            "bots_count": 0,
            "status": "waiting", # waiting, playing, finished
            "players": [
                {"id": host_id, "name": host_name, "is_host": True, 
                 "ready": False, "connected": True}
            ],
            "state": None
        }
        self.rooms[room_id] = room
        return room_id
```

**Room Statuses:**
- `waiting` - Lobby phase, accepting players
- `playing` - Game in progress
- `finished` - Game ended

```python
def get_public_rooms(self):
    public = []
    for r_id, r in self.rooms.items():
        if not r["is_private"] and r["status"] == "waiting":
            current_occupancy = len(r["players"]) + r["bots_count"]
            if current_occupancy < r["total_slots"]:
                public.append({
                    "id": r["id"],
                    "host_name": r["host_name"],
                    "bet_amount": r["bet_amount"],
                    "players": len(r["players"]),
                    "bots": r["bots_count"],
                    "total_slots": r["total_slots"]
                })
    return public
```

<Note>
Only rooms with status `"waiting"` and available slots appear in public listings.
</Note>

### Join Logic

(`moche_engine.py:50-76`):

```python
def join_room(self, room_id, player_id, player_name):
    room = self.get_room(room_id)
    if not room:
        return False, "Sala no encontrada"
        
    if room["status"] != "waiting":
        return False, "La partida ya ha comenzado"

    # Check if player is already in room (reconnection)
    for p in room["players"]:
        if p["id"] == player_id:
            p["connected"] = True
            return True, "Reconectado"

    current_occupancy = len(room["players"]) + room["bots_count"]
    if current_occupancy >= room["total_slots"]:
        return False, "La sala está llena"

    room["players"].append({
        "id": player_id,
        "name": player_name,
        "is_host": False,
        "ready": False,
        "connected": True
    })
    return True, "Unido con éxito"
```

```python
def toggle_ready(self, room_id, player_id):
    room = self.get_room(room_id)
    if not room: return False
    for p in room["players"]:
        if p["id"] == player_id:
            p["ready"] = not p["ready"]
            return True
    return False
```

## Difficulty Levels & Bot Filling

### Bot Count Calculation

When a game starts, bots are added dynamically based on difficulty (`moche_engine.py:110-189`):

```python
def start_game(self, room_id, host_id):
    room = self.get_room(room_id)
    if not room or room["host"] != host_id or room["status"] != "waiting":
        return False, "No puedes iniciar la partida"

    difficulty = room.get("difficulty", "easy")
    real_players_count = len(room["players"])
    max_possible_bots = room["total_slots"] - real_players_count
    
    # Difficulty-based bot generation
    if difficulty == "easy":
        bots_count = random.randint(1, 2)
    elif difficulty == "medium":
        bots_count = random.randint(2, 3)
    else: # hard, pro
        bots_count = 3
        
    # Ensure constraints
    bots_count = min(bots_count, max_possible_bots)
    if real_players_count == 1:
        bots_count = max(1, bots_count)  # At least 1 bot if solo
        
    room["bots_count"] = bots_count
```

### Hiding Opponent Cards

(`moche_engine.py:219-238`):

```python
def get_public_state_for_player(self, room_id, requesting_player_id):
    room = self.get_room(room_id)
    if not room or not room["state"]: return None

    import copy
    state = copy.deepcopy(room["state"])
    
    # Hide opponent card values
    for pid, pdata in state["players"].items():
        if pid != requesting_player_id:
            pdata["cards"] = [{"hidden": True} for c in pdata["cards"]]
    
    return {
        "room_id": room_id,
        "bet_amount": room["bet_amount"],
        "difficulty": room.get("difficulty", "easy"),
        "bots_count": room.get("bots_count", 0),
        "state": state
    }
```

<Warning>
Opponent cards are replaced with `{"hidden": True}` placeholders to prevent cheating.
</Warning>

### Room Cleanup

(`moche_engine.py:77-96`):

```python
def leave_room(self, room_id, player_id):
    room = self.get_room(room_id)
    if not room: return False

    # Remove player
    room["players"] = [p for p in room["players"] if p["id"] != player_id]

    if len(room["players"]) == 0:
        # Delete room if empty
        del self.rooms[room_id]
        return True

    # If host left, reassign host to first remaining player
    if room["host"] == player_id:
        new_host = room["players"][0]
        room["host"] = new_host["id"]
        room["host_name"] = new_host["name"]
        new_host["is_host"] = True

    return True
```

<Note>
Rooms are automatically deleted when the last player leaves, preventing memory leaks.
</Note>
